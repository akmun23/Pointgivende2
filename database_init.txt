create schema company;
use company;

create table task (
  task_id int not null AUTO_INCREMENT,
  description char(125),
  time int not null,
  primary key (task_id)
); 

CREATE TABLE robots (
	robot_id int not null AUTO_INCREMENT, 
	name varchar(255), 
	current_task int,
	primary key (robot_id)
	);
